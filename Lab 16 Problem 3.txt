#include <Arduino.h>

// ===== Joystick =====
const int Y_pin = A1;

// ===== H-Bridge =====
const int enA = 3;  // right motor
const int in1 = 4;
const int in2 = 5;

const int enB = 6;  // left motor
const int in3 = 8;
const int in4 = 7;

int deadzone = 100;

void setup() {
  pinMode(enA, OUTPUT);
  pinMode(in1, OUTPUT);
  pinMode(in2, OUTPUT);

  pinMode(enB, OUTPUT);
  pinMode(in3, OUTPUT);
  pinMode(in4, OUTPUT);

  Serial.begin(115200);
}

void loop() {
  int yVal = analogRead(Y_pin);

  // ===== FORWARD =====
  if (yVal > (512 + deadzone)) {
    int speedVal = map(yVal, 512 + deadzone, 1023, 0, 255);

    // RIGHT motor (unchanged)
    digitalWrite(in1, HIGH);
    digitalWrite(in2, LOW);

    // LEFT motor (FLIPPED)
    digitalWrite(in3, LOW);
    digitalWrite(in4, HIGH);

    analogWrite(enA, speedVal);
    analogWrite(enB, speedVal);
  }

  // ===== BACKWARD =====
  else if (yVal < (512 - deadzone)) {
    int speedVal = map(yVal, 0, 512 - deadzone, 255, 0);

    // RIGHT motor
    digitalWrite(in1, LOW);
    digitalWrite(in2, HIGH);

    // LEFT motor (FLIPPED)
    digitalWrite(in3, HIGH);
    digitalWrite(in4, LOW);

    analogWrite(enA, speedVal);
    analogWrite(enB, speedVal);
  }

  // ===== STOP =====
  else {
    analogWrite(enA, 0);
    analogWrite(enB, 0);
  }

  delay(50);
}
